<script>
		$(document).ready(function(){
			// persist on top
			$pos = $('#persist').offset().top;
			$(window).on('scroll', function(){
				if($(window).scrollTop() >= $pos) {
					$('#persist').addClass('persist');
				} else {
					$('#persist').removeClass('persist');
				}
			});
		});
</script>

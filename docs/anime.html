Css Anime

A start point to build CSS animations. With a small jQuery scripts can be used to build presentations and transitions between panels. Based on Dan Eden's Animate.css, Codrops and ElementTransitions.js.

Create beautiful animations and cool slides with a simple markup and discover the power of jQuery and CSS combination.

<link href="../hipertextos/css/anime.css" rel="stylesheet" media="screen">
<script src="../hipertextos/js/anime.js" type="text/javascript"></script>

On scroll page with jquery

On scroll with jQuery (and re-anime when come back to top).

Sample

.animeonscroll

#animeonscroll

Markup
<!-- Snippets -->
<h4 class="animeinscroll">.animeinscroll</h4>
<h4 id="animeinscroll">#animeinscroll</h4>

<!-- jQuery -->
<script>
	// Animations on scroll
	// Create a window variable to set the scroll
	var win = $(window);
	// Create a variable for .anim
	var variable1 = $('.animeonscroll');
	var variable2 = $('#animeonscroll');
	// Apply the animations on scroll
	win.scroll(function(event) {
	  variable1.each(function(i, el) {
		var el = $(el);
		if (el.visible(true)) {
		  el.addClass('anime anm-moveFromLeftFade slow'); 
		} else {
		  el.removeClass('anime anm-moveFromLeftFade slow'); 
		} 
	  });
	});
	win.scroll(function(event) {
	  variable2.each(function(i, el) {
		var el = $(el);
		if (el.visible(true)) {
		  el.addClass('anime anm-moveFromRightFade slow'); 
		} else {
		  el.removeClass('anime anm-moveFromRightFade slow'); 
		} 
	  });
	});
</script>
